27int validarExpressao(char *exp)
28{
29 if (exp == NULL){ return 0; }
30 Pilha *p = criaPilha();
31 int i = 0;
32 while (exp[i] != '\0'){
33 char caractere = exp[i];
34 if (caractere == '(' || caractere == '[' || caractere == '{'){ // Insere na pilha somente os caracteres de abertura
35 /* Para tratar a opção b, foi feito uso dos valores dos caracteres de delimitação
36 * na tabela ASCII, onde os delimitadores com maior precedência sempre têm valores
37 * númericos menores que os de menor precedência.
38 *
39 * Exemplo: (A{B-C}*E), com '(' estando no topo da pilha, tenta-se empilhar '{',
40 * os respectivos valores desses caracteres em decimal são 40 e 123, então a comparação
41 * feita abaixo fica: 40 - 123 < 0 -> verdadeiro, ou seja, a expressão é inválida pela
42 * opção b, pois o '{' está numa posição mais interna que '('.
43 */
44 if (!estaVazia(p) && (*(char *)p->topo->chave - caractere < 0)){
45 esvaziaPilha(p);
46 free(p);
47 return 0;
48 }
49 char *dadoEmpilhar = (char *)malloc(sizeof(char));
50 if (dadoEmpilhar == NULL){
51 perror("Erro ao alocar dado.\n");
52 exit(1);
53 }
54 *dadoEmpilhar = caractere;
55 empilha(p, dadoEmpilhar);
56 }
57 else if (caractere == ')' || caractere == ']' || caractere == '}'){
58 if (estaVazia(p)){
59 esvaziaPilha(p);
60 free(p);
61 return 0;
62 }
63 char topo = *(char *)desempilha(p);
64
65 if ((caractere == ')' && topo != '(') ||
66 (caractere == ']' && topo != '[') ||
67 (caractere == '}' && topo != '{')){
68 esvaziaPilha(p);
69 free(p);
70 return 0;
71 }
72 }
73 i++;
74 }
75 int valida = estaVazia(p); // A pilha deve estar vazia para garantir que não há delimitadores de abertura sem fechamento
76 esvaziaPilha(p);
77 free(p);
78
79 return valida;
80}
